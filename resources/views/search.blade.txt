    <script>
        document.addEventListener('DOMContentLoaded', function() {
            const governorateSelect = document.getElementById('governorateFilter');
            const regionSelect = document.getElementById('regionFilter');
            const filterForm = document.getElementById('filterForm');
            
            // التحقق من وجود العناصر
            if (!governorateSelect || !regionSelect) {
                console.error('❌ عناصر الفلترة غير موجودة!');
                return;
            }

            // ✅ دالة تحميل المناطق (محسنة بنفس طريقة ملف الادمن)
            function loadRegions(governorateId, selectedRegionId = null) {
                // إذا لم يتم تحديد محافظة، نعرض الرسالة المناسبة
                if (!governorateId || governorateId === '') {
                    regionSelect.innerHTML = '<option value="">اختر المحافظة أولاً</option>';
                    regionSelect.disabled = true;
                    return;
                }

                // عرض حالة التحميل
                regionSelect.innerHTML = '<option value="" disabled>⏳ جاري التحميل...</option>';
                regionSelect.disabled = true;

                // ✅ استخدام نفس مسار API المستخدم في ملف الادمن
                const url = `/dlil/get-regions/${governorateId}`;
                console.log('📡 جاري تحميل المناطق من:', url);

                fetch(url, {
                    headers: {
                        'Accept': 'application/json',
                        'X-Requested-With': 'XMLHttpRequest'
                    }
                })
                .then(response => {
                    if (!response.ok) {
                        throw new Error(`HTTP error! status: ${response.status}`);
                    }
                    return response.json();
                })
                .then(data => {
                    console.log('✅ تم استلام البيانات:', data);
                    
                    // تنظيف الـ select وإضافة الخيار الافتراضي
                    regionSelect.innerHTML = '<option value="">جميع المناطق</option>';
                    
                    if (data && Array.isArray(data) && data.length > 0) {
                        data.forEach(region => {
                            const option = document.createElement('option');
                            option.value = region.id;
                            option.textContent = region.name;
                            if (selectedRegionId && selectedRegionId == region.id) {
                                option.selected = true;
                            }
                            regionSelect.appendChild(option);
                        });
                        regionSelect.disabled = false;
                        console.log(`✅ تم تحميل ${data.length} منطقة`);
                    } else {
                        regionSelect.innerHTML = '<option value="">⚠️ لا توجد مناطق مسجلة لهذه المحافظة</option>';
                        regionSelect.disabled = false;
                    }
                })
                .catch(error => {
                    console.error('❌ خطأ في تحميل المناطق:', error);
                    regionSelect.innerHTML = '<option value="">❌ حدث خطأ في التحميل</option>';
                    regionSelect.disabled = false;
                });
            }

            // ✅ عند تغيير المحافظة (نفس طريقة ملف الادمن)
            governorateSelect.addEventListener('change', function() {
                const governorateId = this.value;
                console.log('🔄 تغيير المحافظة إلى:', governorateId);
                
                // تحميل المناطق مباشرة باستخدام الـ ID
                loadRegions(governorateId);
                
                // إرسال النموذج لتحديث النتائج
                if (filterForm) {
                    setTimeout(() => {
                        filterForm.submit();
                    }, 300);
                }
            });

            // عند تغيير المنطقة - إرسال النموذج
            regionSelect.addEventListener('change', function() {
                if (filterForm) {
                    setTimeout(() => {
                        filterForm.submit();
                    }, 300);
                }
            });

            // ✅ تحميل المناطق عند تحميل الصفحة (نفس طريقة ملف الادمن)
            const initialGovernorate = governorateSelect.value;
            const initialRegion = '{{ request('region') }}';
            
            if (initialGovernorate) {
                loadRegions(initialGovernorate, initialRegion);
            }

            // تحسين تجربة المستخدم - إرسال النموذج عند تغيير أي فلتر
            document.querySelectorAll('#filterForm select, #filterForm input[type="checkbox"]').forEach(el => {
                el.addEventListener('change', function(e) {
                    // منع الإرسال المزدوج للمنطقة والمحافظة (لأننا نرسلهما يدوياً)
                    if (this.id === 'regionFilter' || this.id === 'governorateFilter') {
                        return;
                    }
                    if (filterForm) {
                        setTimeout(() => {
                            filterForm.submit();
                        }, 300);
                    }
                });
            });

            // تحسين البحث - إرسال النموذج عند الضغط على Enter
            document.getElementById('searchInput')?.addEventListener('keypress', function(e) {
                if (e.key === 'Enter') {
                    e.preventDefault();
                    this.closest('form').submit();
                }
            });

            console.log('✅ تم تهيئة نظام البحث بنجاح');
        });
    </script>

    <script>
        // تحميل الصور بتقنية Lazy Loading محسنة
        document.addEventListener('DOMContentLoaded', function() {
            if ('IntersectionObserver' in window) {
                const images = document.querySelectorAll('img[loading="lazy"]');
                const imageObserver = new IntersectionObserver((entries) => {
                    entries.forEach(entry => {
                        if (entry.isIntersecting) {
                            const img = entry.target;
                            img.src = img.src; // Trigger load
                            imageObserver.unobserve(img);
                        }
                    });
                });
                images.forEach(img => imageObserver.observe(img));
            }
        });
    </script>
